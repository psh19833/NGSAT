```python
async def get_volume_rank(self, market: str = "J") -> list[dict]:
    """KIS 거래량순위 API 호출 — 실시간 거래량 상위 종목 리스트.

    Args:
        market: J=KRX, NX=NXT

    Returns:
        [{"code": "005930", "name": "삼성전자", "volume": 12345678}, ...]
    """
    params = {
        "FID_COND_MRKT_DIV_CODE": market,
        "FID_COND_SCR_DIV_CODE": "20171",
        "FID_INPUT_ISCD": "0000",
        "FID_DIV_CLS_CODE": "0",
        "FID_BLNG_CLS_CODE": "0",
        "FID_TRGT_CLS_CODE": "000000000",
        "FID_TRGT_EXLS_CLS_CODE": "0",
        "FID_INPUT_PRICE_1": "",
        "FID_INPUT_PRICE_2": "",
        "FID_VOL_CNT": "",
        "FID_INPUT_DATE_1": datetime.now().strftime("%Y%m%d"),
    }
    resp = await self._http.get("volume_rank", params=params)
    if not resp.success:
        return []
    output = resp.raw.get("output", [])
    return [
        {"code": item["stck_shrn_iscd"], "name": item["hts_kor_isnm"],
         "volume": int(item.get("acml_vol", 0))}
        for item in output if item.get("stck_shrn_iscd")
    ]
```

## Phase 2 — WebSocket 배치 구독/해제 (리스크 LOW, 30분)

### 2-1: 배치 메서드 추가

**파일:** `data/adapters/kis/websocket_client.py`

```python
async def batch_subscribe(self, codes: list[str]) -> None:
    """여러 종목 한 번에 구독."""
    for code in codes:
        self._subscribed.add(code)
    if self._ws and self._running:
        for code in codes:
            await self._send_subscribe(code)

async def batch_unsubscribe(self, codes: list[str]) -> None:
    """여러 종목 한 번에 구독 해제."""
    for code in codes:
        self._subscribed.discard(code)
    if self._ws and self._running:
        for code in codes:
            await self._send_unsubscribe(code)

async def swap_universe(self, add_codes: list[str], remove_codes: list[str]) -> None:
    """유니버스 교체: 구독 해제 → 새 구독."""
    if remove_codes:
        await self.batch_unsubscribe(remove_codes)
    if add_codes:
        await self.batch_subscribe(add_codes)
    logger.info(f"WebSocket 유니버스 교체: -{len(remove_codes)} +{len(add_codes)} = {len(self._subscribed)}종목")
```

## Phase 3 — RealDataProvider 동적 교체 (리스크 ZERO, 1시간)

### 3-1: swap_universe() 메서드

**파일:** `data/real_data_provider.py`

```python
async def swap_universe(self, new_codes: list[str]) -> None:
    """실시간으로 유니버스 종목 교체.

    Args:
        new_codes: 새 유니버스 종목코드 리스트 (최대 40개).
    """
    old_codes = set(self._codes)
    new_set = set(new_codes)
    add_codes = list(new_set - old_codes)
    remove_codes = list(old_codes - new_set)

    if not add_codes and not remove_codes:
        logger.info("유니버스 교체 불필요 (변동 없음)")
        return

    # 1. WebSocket 구독 교체
    if self._ws:
        await self._ws.swap_universe(add_codes, remove_codes)

    # 2. 새 종목 일봉 데이터 로드
    if add_codes:
        adapter = await self._get_adapter()
        for code in add_codes:
            try:
                prices = await adapter.get_daily_chart(code)
                if prices:
                    market = _infer_market(code)
                    self._universe_cache.append(
                        (StockInfo(code=code, name="", market=market), prices)
                    )
                await asyncio.sleep(0.1)  # Rate Limit 보호
            except Exception as e:
                logger.warning(f"[{code}] 신규 편입 일봉 로드 실패: {e}")

    # 3. 제거 종목 캐시 정리
    if remove_codes:
        self._universe_cache = [
            (info, prices) for info, prices in self._universe_cache
            if info.code not in remove_codes
        ]

    # 4. MinuteBarBuilder 리셋 (신규 종목은 새로 수집 시작)
    for code in remove_codes:
        self._minute_builder.clear(code)

    self._codes = new_set
    logger.info(f"유니버스 교체 완료: {len(old_codes)}→{len(new_codes)}종목 "
                f"(+{len(add_codes)}, -{len(remove_codes)})")
```

## Phase 4 — Orchestrator 09:10 지연 + 교체 (리스크 LOW, 2시간)

### 4-1: 진입 가능 시간 체크

**파일:** `live/orchestrator.py`

```python
TRADING_START_TIME = 9  # 시: 09시
TRADING_START_MINUTE = 10  # 분: 10분 → 09:10

def _can_enter_trade(self) -> bool:
    """09:10 이전에는 매매 진입 불가 (데이터 수집 시간)."""
    now = datetime.now(KST)
    return now.hour > TRADING_START_TIME or (
        now.hour == TRADING_START_TIME and now.minute >= TRADING_START_MINUTE
    )
```

**Step 6 (ML 진입)에서 사용:**

### 4-2: 유니버스 교체 트리거

**파일:** `main.py` (trading_loop 내부)

```python
_universe_swapped = False  # 오늘 교체 완료 여부

while True:
    now = datetime.now(KST)
    
    # 09:08~09:09: 유니버스 교체 (1일 1회)
    if not _universe_swapped and now.hour == 9 and now.minute == 9:
        try:
            # 거래량순위 조회
            rank_data = await broker.get_volume_rank()
            top_60 = [item["code"] for item in rank_data[:60]]
            
            # 보유 포지션 유지
            held = {p.code for p in await broker.get_positions()}
            
            # 거래량 상위 35 + 보유포지션 = 40종목
            new_universe = list(dict.fromkeys(top_60))  # 중복 제거
            new_universe = [c for c in new_universe if c not in held][:35]
            new_universe = list(held) + new_universe
            new_universe = new_universe[:40]  # 최대 40
            
            # 교체 실행
            await data_provider.swap_universe(new_universe)
            _universe_swapped = True
            logger.info(f"유니버스 교체 완료: {len(new_universe)}종목")
        except Exception as e:
            logger.warning(f"유니버스 교체 실패 (기존 유지): {e}")
```

`swap_universe()`에서 `held_codes`에 포함된 종목은 절대 제외되지 않도록 보장:

```python
# 보유 포지션은 무조건 유지
remove_codes = [c for c in remove_codes if c not in held_codes]
```
