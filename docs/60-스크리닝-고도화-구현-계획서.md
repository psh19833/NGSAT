#### OBV + 기울기

```python
def obv(closes, volumes) -> np.ndarray:
    """On-Balance Volume."""
    obv_vals = np.zeros(len(closes))
    for i in range(1, len(closes)):
        if closes[i] > closes[i-1]:
            obv_vals[i] = obv_vals[i-1] + volumes[i]
        elif closes[i] < closes[i-1]:
            obv_vals[i] = obv_vals[i-1] - volumes[i]
        else:
            obv_vals[i] = obv_vals[i-1]
    return obv_vals

def obv_slope(obv_vals, period=20) -> float:
    """OBV 20일 선형회귀 기울기."""
    x = np.arange(period)
    y = obv_vals[-period:]
    slope = np.polyfit(x, y, 1)[0]
    return slope
```

**리스크**: 없음 — 순수 계산 함수

#### RS (Relative Strength)

```python
def relative_strength(stock_closes, index_closes, period=20) -> float:
    """종목 수익률 / 지수 수익률."""
    stock_ret = (stock_closes[-1] - stock_closes[-period]) / stock_closes[-period]
    index_ret = (index_closes[-1] - index_closes[-period]) / index_closes[-period]
    return stock_ret / index_ret if index_ret != 0 else 1.0
```

**리스크**: 지수 데이터(`index_prices`) 필요 — `screen_stocks()`에서 이미 사용 중인 `index_prices` 파라미터를 `_evaluate_single_stock()`에 전달해야 함

#### 캔들스틱 패턴

```python
def detect_hammer(open, high, low, close) -> bool:
    """망치형 — 하단 긴 그림자 + 몸통 작음."""
    body = abs(close - open)
    lower_shadow = min(open, close) - low
    upper_shadow = high - max(open, close)
    return lower_shadow >= body * 2 and upper_shadow <= body * 0.3

def detect_engulfing(prev_open, prev_close, open, close) -> bool:
    """장악형 — 양봉이 전일 음봉 완전 덮음."""
    prev_body = abs(prev_close - prev_open)
    body = abs(close - open)
    return (close > open and prev_close < prev_open
            and close > prev_open and open < prev_close)
```
