```python
    """체결강도 상위 — 매수압력 순위."""
    params = {
        "FID_TRGT_EXLS_CLS_CODE": "0",
        "FID_COND_MRKT_DIV_CODE": "J",
        "FID_COND_SCR_DIV_CODE": "20168",
        "FID_INPUT_ISCD": "0000",
        "FID_DIV_CLS_CODE": "0",
        "FID_INPUT_PRICE_1": "",
        "FID_INPUT_PRICE_2": "",
        "FID_VOL_CNT": "",
        "FID_TRGT_CLS_CODE": "0",
    }
    resp = await self._http.get("volume_power", params=params)
    if not resp.success:
        return []
    output = resp.raw.get("output", [])
    return [{"code": item["stck_shrn_iscd"], "name": item["hts_kor_isnm"],
             "score": int(item.get("acml_vol", 0) or 0)}
            for item in output if item.get("stck_shrn_iscd")]
```

## Phase 2 — UniverseManager 신규 생성 (리스크 LOW, 2시간)

### 2-1: 데이터 구조

**신규 파일:** `data/universe_manager.py`

```
UniverseManager:
  ─ active: dict[str, StockInfo]    # 현재 매매 중인 40종목
  ─ reserve: dict[str, StockInfo]   # 예비 60종목
  ─ last_swap: datetime             # 마지막 교체 시간
  ─ held_codes: set[str]            # 보유 포지션 (절대 제외)
```

### 2-2: 핵심 메서드

```python
class UniverseManager:
    async def initialize(self, broker: KisAdapter, provider: RealDataProvider):
        """09:00 초기화 — 3개 순위 API + 4축 점수 → TOP 40 + 예비 60."""
        # 1. multi_rank() 호출 (3 REST)
        volume_rank = await broker.get_volume_rank()        # 100종목
        volume_power = await broker.get_volume_power()      # 100종목
        fluctuation = await broker.get_fluctuation_rank()   # 100종목
        
        # 2. 4축 점수 합산
        all_codes = self._merge_rankings(volume_rank, volume_power, fluctuation)
        scored = await self._score_candidates(all_codes, provider)
        
        # 3. TOP 40 = active, 나머지 60 = reserve
        scored.sort(key=lambda x: x.composite_score, reverse=True)
        self.active = {s.code: s for s in scored[:40]}
        self.reserve = {s.code: s for s in scored[40:]}
        
        # 4. 100종목 일봉 데이터 로드 (100 REST, 100ms 간격)
        await self._load_daily_data(all_codes, provider)
        
        # 5. WebSocket 40종목 구독
        ws_codes = list(self.active.keys())
        await provider._ws.swap_universe(ws_codes, [])
        logger.info(f"유니버스 초기화: active={len(self.active)}, reserve={len(self.reserve)}")

    async def swap(self, broker: KisAdapter, provider: RealDataProvider):
        """5분 교체 — 하위 20 제외 + 상위 20 편입."""
        # 1. multi_rank() 재호출 (3 REST)
        volume_rank = await broker.get_volume_rank()
        volume_power = await broker.get_volume_power()
        fluctuation = await broker.get_fluctuation_rank()
        
        # 2. 현재 active 40종목 재평가
        active_list = list(self.active.values())
        rescored = await self._score_candidates(
            [s.code for s in active_list], provider, volume_rank, volume_power, fluctuation
        )
        
        # 3. 하위 20 선정 (보유포지션 제외)
        rescored.sort(key=lambda x: x.composite_score)
        to_remove = [s for s in rescored if s.code not in self.held_codes][:20]
        remove_codes = {s.code for s in to_remove}
        
        # 4. 예비 60 + 신규 100 → 상위 20 편입
        candidate_codes = list(self.reserve.keys())
        candidate_codes += [item["code"] for item in volume_rank[:100]
                          if item["code"] not in self.active and item["code"] not in self.reserve]
        candidates = await self._score_candidates(
            candidate_codes, provider, volume_rank, volume_power, fluctuation
        )
        candidates.sort(key=lambda x: x.composite_score, reverse=True)
        to_add = candidates[:20]
        
        # 5. 교체 실행
        for s in to_add:
            if s.code not in self.active:
                self.active[s.code] = s
        for s in to_remove:
            self.active.pop(s.code, None)
            self.reserve[s.code] = s
        
        # 6. 예비 리스트 갱신
        reserve_new = list(self.reserve.keys()) + [s.code for s in active_list if s.code in remove_codes]
        self.reserve = {c: self.reserve.get(c) for c in reserve_new[:60] if c not in self.active}
        
        # 7. WebSocket 구독 교체
        await provider._ws.swap_universe(
            [s.code for s in to_add],
            [s.code for s in to_remove]
        )
```

## Phase 3 — main.py 통합 (리스크 LOW, 1시간)

### 3-1: 기존 main.py 교체 로직 대체

**파일:** `main.py` (trading_loop 내부)

기존에 구현한 `_universe_swapped` + `volume_rank` 호출 로직을 `UniverseManager`로 대체:

```python
# main.py trading_loop 내부
if not hasattr(orchestrator, '_universe_manager'):
    orchestrator._universe_manager = UniverseManager()
    orchestrator._universe_manager.held_codes = set()  # updated each cycle

# 09:00 ~ 09:10: 초기화 + 09:10부터 매매
if not orchestrator._universe_manager.initialized:
    # 09:00~09:10: 데이터 수집
    if kst_now.hour == 9 and kst_now.minute >= 0 and kst_now.minute < 10:
        if kst_now.minute == 0 and kst_now.second < 30:
            await orchestrator._universe_manager.initialize(broker, data_provider)
        orchestrator._trading_allowed = False
    else:
        orchestrator._trading_allowed = True
        orchestrator._universe_manager.initialized = True

# 5분 교체
elif orchestrator._universe_manager.should_swap(kst_now):
    await orchestrator._universe_manager.swap(broker, data_provider)

# 보유 포지션 업데이트
held = {p.code for p in (await broker.get_positions())}
orchestrator._universe_manager.held_codes = held
```

## 롤백 계획

| 상황 | 롤백 방법 |
|:----|:----------|
| 체결강도/등락률 API 실패 | endpoint 제거, 기존 volume_rank만 사용 |
| UniverseManager 오류 | `orchestrator._universe_manager = None` → 기존 고정 유니버스 |
| 5분 교체 이상 | `UniverseManager.swap()`에서 예외 발생 시 교체 스킵, 다음 주기 재시도 |
| WebSocket 교체 실패 | 기존 구독 유지, 로그만 출력 |
